# Diretorios
~ mkdir 	# cria um novo diretorio
~ rmdir 	# remove um diretorio

# Arquivos
~ find 		# procura arquivos
~ wc 		# conta o numero de linhas, caracteres e palavras do arquivo

# Log's
~ $? 		# mostra o valor de saida do comando
~ echo $SHELL 	# mostra caminho padrao execução scripts

# Senhas
~ cat /etc/passwd 	# arquivo que contem o nome e login dos usuarios do sistema
~ cat /etc/shadow 	# arquivo que contem as senhas dos usuarios do sistema

# Outros
~
